import numpy as np
x=np.array([1,2,3,4])
y=x*2
plt.plot(x,y,ls="solid",lw=6,c="r",marker="o",mec="green",mfc="yellow",ms=10)

plt.ylabel("Y-axis")
plt.title("Line Graph")
