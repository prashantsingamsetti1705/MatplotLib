#Draw the Line Graph for all three weeks
import pandas as pd
from matplotlib import pyplot as plt
df=pd.DataFrame({"Days":["Monday","Tuesday","Wednesday","Thursday","Friday","Saturday","Sunday"],
                 "Week1":[27,29,32,33,33,31,29],
                 "Week2":[29,30,31,33,34,34,31],
                 "Week3":[29,28,25,27,31,30,29]})
print("------------------------------------")
print(df)
print("------------------------------------")
plt.plot(df["Days"],df["Week1"],lw=6,c="red",marker="o")
plt.plot(df["Days"],df["Week2"],lw=6,c="green",marker="X")
plt.plot(df["Days"],df["Week3"],lw=6,c="black",marker="*")
plt.xlabel("WEEK DAYS",fontdict={"family":"Arial","color":"blue","size":15})
plt.ylabel("TEMP",fontdict={"family":"Arial","color":"blue","size":15})
plt.title("TEMP in 3 WEEK DAYS",fontdict={"family":"Arial","color":"blue","size":20})
plt.legend(labels=["Week1","Week2","Week3"])
plt.grid(axis="both",ls="dotted",c="green",lw=2)
plt.show()
