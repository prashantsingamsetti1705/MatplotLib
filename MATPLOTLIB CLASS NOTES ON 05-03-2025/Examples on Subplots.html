import matplotlib.pyplot as plt
plt.subplot(1,2,1)
plt.plot([1,2,3,4],[4,3,2,1])
plt.xlabel("x-axis")
plt.ylabel("y-axis")
plt.title("XY Plot-1")

plt.ylabel("y-axis")

#---------------------------------
plt.suptitle("Main Plot contains subplots ")
plt.show()

import matplotlib.pyplot as plt
import numpy as np

#plot 1:
x = np.array([0, 1, 2, 3])

plt.subplot(1, 2, 1)
plt.plot(x,y)
plt.title("SALES")
